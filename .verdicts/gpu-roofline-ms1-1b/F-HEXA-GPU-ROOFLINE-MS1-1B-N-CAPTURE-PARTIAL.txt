Captures the `;`-size token via `p_advance().value` (the parser's standard token accessor):
    let mut _size = ""
    if p_peek_kind() == "Semicolon" {
        p_advance()
        if p_peek_kind() != "RBracket" { _size = p_advance().value }
    }
    p_expect("RBracket")
    if _size != "" { base = "[" + inner + "; " + _size + "]" }   // [T; N] carries N
    else           { base = "[" + inner + "]" }                  // plain [T] unchanged
